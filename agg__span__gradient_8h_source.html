88  //--------------------------------------------------------------------
89  void generate(color_type* span, int x, int y, unsigned len)
90  {
91  int dd = m_d2 - m_d1;
92  if(dd < 1) dd = 1;
93  m_interpolator->begin(x+0.5, y+0.5, len);
94  do
95  {
96  m_interpolator->coordinates(&x, &y);
97  int d = m_gradient_function->calculate(x >> downscale_shift,
98  y >> downscale_shift, m_d2);
99  d = ((d - m_d1) * (int)m_color_function->size()) / dd;
100  if(d < 0) d = 0;
101  if(d >= (int)m_color_function->size()) d = m_color_function->size() - 1;
102  *span++ = (*m_color_function)[d];
103  ++(*m_interpolator);
104  }
105  while(--len);
106  }

186  class gradient_radial_focus
187  {
188  public:
189  //---------------------------------------------------------------------
190  gradient_radial_focus() :

199  gradient_radial_focus(double r, double fx, double fy) :
200  m_r (iround(r * gradient_subpixel_scale)),
201  m_fx(iround(fx * gradient_subpixel_scale)),
202  m_fy(iround(fy * gradient_subpixel_scale))
203  {
204  update_values();
205  }
206 
207  //---------------------------------------------------------------------
208  void init(double r, double fx, double fy)
209  {
210  m_r = iround(r * gradient_subpixel_scale);
211  m_fx = iround(fx * gradient_subpixel_scale);
212  m_fy = iround(fy * gradient_subpixel_scale);
213  update_values();
214  }
215 
216  //---------------------------------------------------------------------
217  double radius() const { return double(m_r) / gradient_subpixel_scale; }
218  double focus_x() const { return double(m_fx) / gradient_subpixel_scale; }
219  double focus_y() const { return double(m_fy) / gradient_subpixel_scale; }
220 
221  //---------------------------------------------------------------------
222  int calculate(int x, int y, int) const
223  {
224  double dx = x - m_fx;
225  double dy = y - m_fy;
226  double d2 = dx * m_fy - dy * m_fx;
227  double d3 = m_r2 * (dx * dx + dy * dy) - d2 * d2;
228  return iround((dx * m_fx + dy * m_fy + sqrt(fabs(d3))) * m_mul);
229  }
230 
231  private:
232  //---------------------------------------------------------------------
233  void update_values()
234  {
235  // Calculate the invariant values. In case the focal center
236  // lies exactly on the gradient circle the divisor degenerates
237  // into zero. In this case we just move the focal center by
238  // one subpixel unit possibly in the direction to the origin (0,0)
239  // and calculate the values again.
240  //-------------------------
241  m_r2 = double(m_r) * double(m_r);
242  m_fx2 = double(m_fx) * double(m_fx);
243  m_fy2 = double(m_fy) * double(m_fy);
244  double d = (m_r2 - (m_fx2 + m_fy2));
245  if(d == 0)
246  {
247  if(m_fx) { if(m_fx < 0) ++m_fx; else --m_fx; }
248  if(m_fy) { if(m_fy < 0) ++m_fy; else --m_fy; }
249  m_fx2 = double(m_fx) * double(m_fx);
250  m_fy2 = double(m_fy) * double(m_fy);
251  d = (m_r2 - (m_fx2 + m_fy2));
252  }
253  m_mul = m_r / d;
254  }

257  int m_fx;
258  int m_fy;
259  double m_r2;
260  double m_fx2;
261  double m_fy2;
262  double m_mul;
263  };
